function resizeGridA4()
{
  width = 100 * 3.779528;
  height = 150 * 3.779528;
  
  width_Grid_px  	= $("#grilleContainer").width();
  height_Grid_px 	= $("#grilleContainer").height();
  
  ratio  = Math.min(width/width_Grid_px,height/height_Grid_px);
  $("#grilleContainer").attr("style","transform : scale("+ratio+");");
  
  //.clone().appendTo( ".goodbye" );
}

function beforePrint(){
    resizeGridA4();
}
    
function afterPrint() {
    $("#grilleContainer").attr("style","transform : scale(1);");
}
    
    
    if (window.matchMedia) {
        var mediaQueryList = window.matchMedia('print');
        mediaQueryList.addListener(function(mql) {
            if (mql.matches) {
                beforePrint();
            } else {
                afterPrint();
            }
        });
    }
    
    window.onbeforeprint = beforePrint;
    window.onafterprint = afterPrint;
});
